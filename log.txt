Generate JWT + get access token
import jwt
import time
import requests

INTEGRATION_KEY = "your_client_id"
USER_ID = "your_user_guid"
ACCOUNT_ID = "your_account_id"
PRIVATE_KEY = open("private.key").read()

def get_access_token():
    now = int(time.time())

    payload = {
        "iss": INTEGRATION_KEY,
        "sub": USER_ID,
        "aud": "account-d.docusign.com",
        "iat": now,
        "exp": now + 3600,
        "scope": "signature impersonation"
    }

    token = jwt.encode(payload, PRIVATE_KEY, algorithm="RS256")

    res = requests.post(
        "https://account-d.docusign.com/oauth/token",
        data={
            "grant_type": "urn:ietf:params:oauth:grant-type:jwt-bearer",
            "assertion": token
        }
    )

    return res.json()["access_token"]
Step 5 — Use token (send envelope)
def send_envelope(access_token):
    url = f"https://demo.docusign.net/restapi/v2.1/accounts/{ACCOUNT_ID}/envelopes"

    headers = {
        "Authorization": f"Bearer {access_token}",
        "Content-Type": "application/json"
    }

    body = {
        "emailSubject": "Please sign",
        "documents": [
            {
                "documentBase64": "BASE64_PDF",
                "name": "contract.pdf",
                "documentId": "1"
            }
        ],
        "recipients": {
            "signers": [
                {
                    "email": "client@mail.com",
                    "name": "Client Name",
                    "recipientId": "1"
                }
            ]
        },
        "status": "sent"
    }

    res = requests.post(url, headers=headers, json=body)
    return res.json()
